---
title: Configure an allowed address pair
sidebarTitle: Allowed address pair
ai-navigation: Assign additional IP addresses to a Virtual Machine's private network interface port via the allowed address pairs API, enabling traffic routing to multiple IPs on a single MAC address with support for up to 10 pairs per port.
---

An allowed address pair binds an extra IP address — or a CIDR range — to a Virtual Machine's network interface. By default, each port enforces a strict MAC-to-IP binding, so only traffic addressed to the port's own IP is accepted. Adding an entry to the allowed address pairs list lifts that restriction for the specified address, making the machine accessible on both its original IP and the additional one.

A common use case is distributing a [virtual IP address](/cloud/networking/ip-address/create-and-configure-a-virtual-ip-vip-address) across multiple VMs using Keepalived — each VM's port must allow the shared VIP address.

<Info>
A permanent [API token](/account-settings/api-tokens) is required, along with a [project ID](https://api.gcore.com/docs/cloud#tag/Projects/operation/ProjectsListV1.get), a [region ID](https://api.gcore.com/docs/cloud#tag/Regions/operation/RegionListV1.get), and the `port_id` of a private network interface — the external (public) interface does not support allowed address pairs.
</Info>

Set these environment variables before running the examples:

```bash
export GCORE_API_KEY="{YOUR_API_KEY}"
export PROJECT_ID="{YOUR_PROJECT_ID}"
export REGION_ID="{YOUR_REGION_ID}"
export INSTANCE_ID="{YOUR_INSTANCE_ID}"
export PORT_ID="{YOUR_PORT_ID}"        # private interface port ID — see Step 1
```

## Quickstart

The scripts below set two allowed address pairs on a private interface port, poll the task until it completes, verify the result, then remove all pairs.

<Tabs>
  <Tab title="Python">
    ```python
    import os
    import time
    import requests

    API_KEY = os.environ["GCORE_API_KEY"]
    PROJECT_ID = os.environ["PROJECT_ID"]
    REGION_ID = os.environ["REGION_ID"]
    INSTANCE_ID = os.environ["INSTANCE_ID"]
    PORT_ID = os.environ["PORT_ID"]

    session = requests.Session()
    session.headers.update({"Authorization": f"APIKey {API_KEY}"})
    base = f"https://api.gcore.com/cloud"

    # Step 1. Set allowed address pairs
    resp = session.put(
        f"{base}/v2/ports/{PROJECT_ID}/{REGION_ID}/{PORT_ID}/allow_address_pairs",
        json={
            "allowed_address_pairs": [
                {"ip_address": "192.168.123.20", "mac_address": "00:16:3e:f2:87:16"},
                {"ip_address": "192.168.0.0/24"},
            ]
        },
    )
    resp.raise_for_status()
    task_id = resp.json()["tasks"][0]
    print(f"Task: {task_id}")

    # Step 2. Poll until FINISHED
    while True:
        state = session.get(f"{base}/v1/tasks/{task_id}").json()["state"]
        if state == "FINISHED":
            break
        time.sleep(5)
    print("Pairs applied.")

    # Step 3. Verify
    ifaces = session.get(
        f"{base}/v1/instances/{PROJECT_ID}/{REGION_ID}/{INSTANCE_ID}/interfaces"
    ).json()["results"]
    for iface in ifaces:
        if iface["port_id"] == PORT_ID:
            for pair in iface["allowed_address_pairs"]:
                print(f"  ip={pair['ip_address']}  mac={pair['mac_address']}")

    # Step 4. Remove all pairs
    resp = session.put(
        f"{base}/v2/ports/{PROJECT_ID}/{REGION_ID}/{PORT_ID}/allow_address_pairs",
        json={"allowed_address_pairs": []},
    )
    resp.raise_for_status()
    task_id = resp.json()["tasks"][0]
    while True:
        state = session.get(f"{base}/v1/tasks/{task_id}").json()["state"]
        if state == "FINISHED":
            break
        time.sleep(5)
    print("Pairs removed.")
    ```
  </Tab>
  <Tab title="Go">
    ```go
    package main

    import (
        "bytes"
        "encoding/json"
        "fmt"
        "io"
        "log"
        "net/http"
        "os"
        "time"
    )

    var (
        apiKey     = os.Getenv("GCORE_API_KEY")
        projectID  = os.Getenv("PROJECT_ID")
        regionID   = os.Getenv("REGION_ID")
        instanceID = os.Getenv("INSTANCE_ID")
        portID     = os.Getenv("PORT_ID")
        base       = "https://api.gcore.com/cloud"
    )

    func do(method, url string, body any) map[string]any {
        var r io.Reader
        if body != nil {
            b, _ := json.Marshal(body)
            r = bytes.NewReader(b)
        }
        req, _ := http.NewRequest(method, url, r)
        req.Header.Set("Authorization", "APIKey "+apiKey)
        if body != nil {
            req.Header.Set("Content-Type", "application/json")
        }
        resp, err := http.DefaultClient.Do(req)
        if err != nil {
            log.Fatal(err)
        }
        defer resp.Body.Close()
        var result map[string]any
        json.NewDecoder(resp.Body).Decode(&result)
        return result
    }

    func pollTask(taskID string) {
        for {
            r := do("GET", fmt.Sprintf("%s/v1/tasks/%s", base, taskID), nil)
            if r["state"] == "FINISHED" {
                return
            }
            time.Sleep(5 * time.Second)
        }
    }

    func main() {
        // Step 1. Set allowed address pairs
        result := do("PUT",
            fmt.Sprintf("%s/v2/ports/%s/%s/%s/allow_address_pairs", base, projectID, regionID, portID),
            map[string]any{
                "allowed_address_pairs": []map[string]any{
                    {"ip_address": "192.168.123.20", "mac_address": "00:16:3e:f2:87:16"},
                    {"ip_address": "192.168.0.0/24"},
                },
            },
        )
        tasks := result["tasks"].([]any)
        taskID := tasks[0].(string)
        fmt.Printf("Task: %s\n", taskID)

        // Step 2. Poll until FINISHED
        pollTask(taskID)
        fmt.Println("Pairs applied.")

        // Step 3. Verify
        ifaces := do("GET",
            fmt.Sprintf("%s/v1/instances/%s/%s/%s/interfaces", base, projectID, regionID, instanceID),
            nil,
        )
        for _, iface := range ifaces["results"].([]any) {
            m := iface.(map[string]any)
            if m["port_id"] == portID {
                for _, pair := range m["allowed_address_pairs"].([]any) {
                    p := pair.(map[string]any)
                    fmt.Printf("  ip=%s  mac=%s\n", p["ip_address"], p["mac_address"])
                }
            }
        }

        // Step 4. Remove all pairs
        result = do("PUT",
            fmt.Sprintf("%s/v2/ports/%s/%s/%s/allow_address_pairs", base, projectID, regionID, portID),
            map[string]any{"allowed_address_pairs": []any{}},
        )
        tasks = result["tasks"].([]any)
        pollTask(tasks[0].(string))
        fmt.Println("Pairs removed.")
    }
    ```

### Step 1. Find the port ID of a private interface

Each VM network interface has a `port_id`. Allowed address pairs are applied per port, and only ports on private (non-external) networks support this feature.

<Tabs>
  <Tab title="Python">
    ```python
    import os
    import requests

    session = requests.Session()
    session.headers.update({"Authorization": f"APIKey {os.environ['GCORE_API_KEY']}"})

    resp = session.get(
        f"https://api.gcore.com/cloud/v1/instances"
        f"/{os.environ['PROJECT_ID']}/{os.environ['REGION_ID']}"
        f"/{os.environ['INSTANCE_ID']}/interfaces"
    )
    resp.raise_for_status()
    for iface in resp.json()["results"]:
        print(f"port_id={iface['port_id']}  ip={iface['ip_assignments'][0]['ip_address']}")
    ```

Save the `port_id` of the private interface (the one with a private IP range such as `10.x.x.x` or `192.168.x.x`) as `PORT_ID`. The external interface with a public IP does not support allowed address pairs.

### Step 2. Set allowed address pairs

The `PUT` request replaces the entire list of allowed address pairs for the port. To add one pair without losing existing ones, include all current pairs in the request body alongside the new entry.

| Parameter | Required | Description |
|-----------|----------|-------------|
| `ip_address` | Yes | IP address or CIDR range to allow (e.g. `192.168.1.10` or `10.0.0.0/24`) |
| `mac_address` | No | MAC address to associate; defaults to the port's own MAC when omitted |

Up to 10 pairs are allowed per port.

<Tabs>

    ```python
                {"ip_address": "192.168.123.20", "mac_address": "00:16:3e:f2:87:16"},

    resp.raise_for_status()
    ```

## Remove allowed address pairs

To remove all allowed address pairs from a port, send the same `PUT` request with an empty `allowed_address_pairs` array. The operation is asynchronous — poll the returned task until `state` is `FINISHED`.

<Tabs>
  <Tab title="Python">
    ```python
    resp = session.put(
        f"https://api.gcore.com/cloud/v2/ports"
        f"/{os.environ['PROJECT_ID']}/{os.environ['REGION_ID']}"
        f"/{os.environ['PORT_ID']}/allow_address_pairs",
        json={"allowed_address_pairs": []},
    )
    resp.raise_for_status()
    task_id = resp.json()["tasks"][0]
    # poll until FINISHED (same loop as Step 2)
    ```
  </Tab>
  <Tab title="Go">
    ```go
    result := do("PUT",
        fmt.Sprintf("%s/v2/ports/%s/%s/%s/allow_address_pairs", base, projectID, regionID, portID),
        map[string]any{"allowed_address_pairs": []any{}},
    )
    tasks := result["tasks"].([]any)
    pollTask(tasks[0].(string))
    fmt.Println("Pairs removed.")
    ```
  </Tab>
  <Tab title="curl">
    ```bash
    curl -X PUT "https://api.gcore.com/cloud/v2/ports/$PROJECT_ID/$REGION_ID/$PORT_ID/allow_address_pairs" \
      -H "Authorization: APIKey $GCORE_API_KEY" \
      -H "Content-Type: application/json" \
      -d '{"allowed_address_pairs": []}'
    ```

    Response:

    ```json
    {
      "tasks": ["62f3e5e4-c0db-4473-832d-f72b001ad8c2"]
    }
    ```
  </Tab>
</Tabs>

To remove individual pairs rather than all of them, omit only the entries to remove and include the rest in the request body.
